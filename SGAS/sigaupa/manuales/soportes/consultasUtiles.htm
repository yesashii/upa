Determinar el número de campos  por tablas:

select table_name,

       Count(*) as NumCampos

from   information_schema.columns

group  by table_name

order  by table_name

 

Determinar que tablas tienen el mismo campo:

select *

from   information_schema.columns

where  column_name = 'nombreCampo'

 

Determinar el tamaño en espacio de una tabla:

set nocount on 

declare @cmdstr varchar(100) 
declare @sort bit 

select @sort = 1 

/* 
0-> Ordenado por Nombre de Tabla 
1-> Ordenado por tamaño total usado por tabla 
*/ 
create table #tablatemp 
  ( 
     nombretabla         varchar(50), 
     cantidadfilas       int, 
     tamaniotabla        varchar(50), 
     espaciousado        varchar(50), 
     espacioindicesusado varchar(50), 
     espacionousado      varchar(50) 
  ) 

select @cmdstr = 'sp_msforeachtable ''sp_spaceused "?"''' 

insert into #tablatemp 
exec(@cmdstr) 

if @sort = 0 
  select * 
  from   #tablatemp 
  order  by nombretabla 
else 
  select * 
  from   #tablatemp 
  order  by tamaniotabla desc 

drop table #tablatemp 

 

Saber collation de una tabla:

-- para una base de datos 
select Databasepropertyex('sigaupa_test', 'Collation') 

-- para una tabla en donde carreras es el nombre de la tabla  
select name, 
       collation_name 
from   sys.columns 
where  object_id in (select object_id 
                     from   sys.objects 
                     where  name = 'carreras') 

 

Sacar la clave de un usuario por nombre:

--versón nombre y apellido-->

declare @nombre varchar(30);

declare @apellidop varchar(30);

 

set @nombre = '%[string nombre]%';

set @apellidop = '%[string apellido]%'

 

select pers_ncorr,

       protic.obtener_rut(pers_ncorr) as rut,

       pers_tnombre                   as nombre,

       pers_tape_paterno              as apellido_paterno,

       pers_tape_materno              as apellino_materno

from   personas

where  pers_tnombre like @nombre

       and pers_tape_paterno like @apellidop;

--versón nombre y apellido--<

--**************************************************************************--

-- Con el pers_ncorr obtenido en la primera consulta ejecuta lo siguiente **--

--**************************************************************************--

--sacando el usuario y contraseña-->

declare @pers_ncorr int;

 

set @pers_ncorr = [int pers_ncorr];

 

select susu_tlogin                                       as usuario,

       susu_tclave                                       as contasenia,

       protic.obtener_rut(a.pers_ncorr)                  as rut,

       protic.obtener_nombre_completo(a.pers_ncorr, 'n') as nombre

from   sis_usuarios as a

       inner join personas as b

               on a.pers_ncorr = b.pers_ncorr

where  a.pers_ncorr = @pers_ncorr;

--sacando el usuario y contraseña--<
